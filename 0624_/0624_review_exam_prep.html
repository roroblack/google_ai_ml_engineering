핵심 SQL 패턴

기본 모델 생성

CREATE OR REPLACE MODEL `bracketology.ncaa_model`
OPTIONS(model_type='logistic_reg') AS
SELECT season, 'win' AS label, win_seed AS seed, win_school_ncaa AS school_ncaa,
       lose_seed AS opponent_seed, lose_school_ncaa AS opponent_school_ncaa
FROM `bigquery-public-data.ncaa_basketball.mbb_historical_tournament_games`
WHERE season <= 2017
UNION ALL
SELECT season, 'loss' AS label, lose_seed AS seed, lose_school_ncaa AS school_ncaa,
       win_seed AS opponent_seed, win_school_ncaa AS opponent_school_ncaa
FROM `bigquery-public-data.ncaa_basketball.mbb_historical_tournament_games`
WHERE season <= 2017;

2018 예측/오답 분석

CREATE OR REPLACE TABLE `bracketology.predictions` AS
SELECT *
FROM ML.PREDICT(
  MODEL `bracketology.ncaa_model`,
  (SELECT * FROM `data-to-insights.ncaa.2018_tournament_results`)
);

SELECT model.label AS predicted_label, model.prob AS confidence,
       predictions.label AS correct_label, game_date, round, seed, school_ncaa
FROM `bracketology.predictions` AS predictions,
UNNEST(predicted_label_probs) AS model
WHERE model.prob > .8 AND predicted_label <> predictions.label;

업데이트 모델 생성

CREATE OR REPLACE MODEL `bracketology.ncaa_model_updated`
OPTIONS(model_type='logistic_reg') AS
SELECT season, label, poss_40min, pace_rank, pace_rating,
       opp_poss_40min, opp_pace_rank, opp_pace_rating,
       pace_rank_diff, pace_stat_diff, pace_rating_diff,
       pts_100poss, efficiency_rank, efficiency_rating,
       opp_pts_100poss, opp_efficiency_rank, opp_efficiency_rating,
       eff_rank_diff, eff_stat_diff, eff_rating_diff
FROM `bracketology.training_new_features`
WHERE season BETWEEN 2014 AND 2017;

2019 미래 예측

CREATE OR REPLACE TABLE `bracketology.ncaa_2019_tournament_predictions` AS
SELECT *
FROM ML.PREDICT(
  MODEL `bracketology.ncaa_model_updated`,
  (SELECT * FROM `bracketology.ncaa_2019_tournament`)
);

SELECT p.label AS prediction, ROUND(p.prob,3) AS confidence,
       school_ncaa, seed, opponent_school_ncaa, opponent_seed
FROM `bracketology.ncaa_2019_tournament_predictions`,
UNNEST(predicted_label_probs) AS p
WHERE p.prob >= .5 AND school_ncaa = 'Duke'
ORDER BY seed, opponent_seed;

시험 대비 퀴즈

Q1. BigQuery ML에서 label 컬럼이 중요한 이유는?
모델이 예측해야 할 정답 컬럼을 지정하기 때문이다. NCAA 실습에서는 win/loss가 label이다.
Q2. 승자 행과 패자 행을 UNION ALL로 모두 만드는 이유는?
한 경기를 양쪽 팀 관점으로 만들어 이진 분류 학습 데이터를 균형 있게 구성하기 위해서다.
Q3. ML.WEIGHTS에서 낮은 seed의 weight가 큰 것은 무엇을 의미하는가?
낮은 시드가 승리에 양의 영향을 준다는 관계를 모델이 데이터에서 학습했다는 뜻이다.
Q4. ML.EVALUATE와 ML.PREDICT의 차이는?
EVALUATE는 정답이 있는 데이터로 성능 지표를 계산하고, PREDICT는 새 입력에 예측 label과 확률을 붙인다.
Q5. 업데이트 모델에서 팀 이름과 seed를 제외한 이유는?
과거 강팀이나 시드를 외우는 모델을 피하고 pace/efficiency 지표의 일반화 효과를 보기 위해서다.
Q6. 2019 예측에서 label을 NULL로 둔 이유는?
미래 경기는 아직 실제 승패가 없으므로 정답 없이 피처만 구성해 예측 대상으로 사용한다.
